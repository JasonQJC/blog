package createanddestory;

  import java.util.ArrayList;
  import java.util.List;
  import java.util.Objects;
  import java.util.function.Function;
  import java.util.stream.Collectors;
  
  public final class Unchecked {
    @FunctionalInterface
    interface ExceptionFunction {
      R apply(T t) throws Exception;
    }
  
    /**
     * exception catcher need use e.getCause() to get original exception
     */
    public static  Function wrap(ExceptionFunction mapper) {
      Objects.requireNonNull(mapper);
      return t -> {
        try {
          return mapper.apply(t);
        } catch (Exception e) {
          throw new RuntimeException(e);
        }
      };
    }
  
    public static  Function wrap2(ExceptionFunction mapper) {
      Objects.requireNonNull(mapper);
      return t -> {
        try {
          return mapper.apply(t);
        } catch (Exception e) {
          // !!! notice here
          throwException(e);
        }
        return null;
      };
    }
  
    /**
     * exception catcher don't need use e.getCause() to get original exception
     */
    @SuppressWarnings("unchecked")
    public static  void throwException(Exception e) throws T {
      throw (T) e;
    }
  
    /**
     * One of the important thoughts in Functional propgramming is function is pure
     * and no side effects. So lambda don't throw exception, they deal with
     * exceptions by themselves but the code will looks ugly in java like this
     */
    @SuppressWarnings("rawtypes")
    public static List findClasses1(List names) {
      return names.stream().map(className -> {
        try {
          return Class.forName(className);
        } catch (ClassNotFoundException e) {
          e.printStackTrace();
        }
        return null;
      }).collect(Collectors.toList());
    }
  
    /**
     * code looks good
     */
    @SuppressWarnings("rawtypes")
    public static List findClasses2(List names) {
      return names.stream().map(Unchecked.wrap(className -> Class.forName(className))).collect(Collectors.toList());
    }
  
    @SuppressWarnings("rawtypes")
    public static List findClasses3(List names) {
      return names.stream().map(Unchecked.wrap2(className -> Class.forName(className))).collect(Collectors.toList());
    }
  
    public static void main(String[] args) {
      try {
        findClasses2(new ArrayList());
      } catch (Exception e) {
        // get original exception in this way
        e.getCause().printStackTrace();
      }
  
      try {
        findClasses3(new ArrayList());
      } catch (Exception e) {
        // don't need e.getCause()
        e.printStackTrace();
      }
  
    }
  
  }
